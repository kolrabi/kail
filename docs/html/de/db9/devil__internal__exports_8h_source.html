83 #define SIOopenRO(io, f) ((io)->openReadOnly ? (io)->openReadOnly((io)->handle) : NULL)
84 #define SIOopenWR(io, f) ((io)->openWrite ? (io)->openWrite ((io)->handle) : NULL)
85 #define SIOclose( io ) { if ((io)->close) (io)->close((io)->handle); (io)->handle = NULL; }
86 #define SIOread( io, p, s, n) (io)->read ((io)->handle, (p), (s), (n))
87 #define SIOseek( io, s, w) (io)->seek ((io)->handle, (s), (w))

89 #define SIOgetc( io ) (io)->getchar((io)->handle )
90 #define SIOtell( io ) (io)->tell ((io)->handle )
91 #define SIOputc( io, c) (io)->putchar((c), (io)->handle )
92 #define SIOwrite( io, p, s, n) (io)->write ((p), (s), (n), (io)->handle)
93 #define SIOputs( io, s) SIOwrite(io, s, strlen(s), 1)
