97 INLINE ILHANDLE SIOopenRO(SIO *io, ILconst_string FileName) {
98  if (!io || !FileName) return NULL;
99  if (!io->openReadOnly) return NULL;
100 
101  io->handle = io->openReadOnly(FileName);
102  if (io->handle && io->tell && io->seek) {
103  ILuint OrigPos = SIOtell(io);
104  SIOseek(io, 0, SEEK_END);
105  io->lumpSize = SIOtell(io);
106  SIOseek(io, OrigPos, SEEK_SET);
107  } else {
108  io->lumpSize = ~0;
109  }
110  return io->handle;
111 }
112 
116 INLINE ILboolean SIOeof(SIO *io) {
117  if (!io) return IL_TRUE;
118 
119  if (io->lumpSize != (ILuint)~0 && io->tell) {
120  return SIOtell(io) >= io->lumpSize;
121  }
122 
123  if (io->eof) {
124  return io->eof(io->handle);
125  }
126 
127  return IL_TRUE;
128 }
