70  __asm {

91 INLINE void iSwapUInt(ILuint *i) {
92  #ifdef USE_WIN32_ASM
93  __asm {
94  mov ebx, i

96  bswap eax

105  #endif //GCC_X86_ASM
106  #endif //USE_WIN32_ASM
107 }

117 INLINE void iSwapDouble(ILdouble *d) {
118  #ifdef GCC_X86_ASM
119  int *t = (int*)d;
120  asm("bswap %2 \n"
121  "bswap %3 \n"
122  "movl %2,%1 \n"
123  "movl %3,%0 \n"
124  : "=g" (t[0]), "=g" (t[1])
125  : "r" (t[0]), "r" (t[1]));
126  #else
127  ILubyte t,*b = (ILubyte*)d;
128  #define dswap(x,y) t=b[x]; b[x]=b[y]; b[y]=t;
129  dswap(0,7);
130  dswap(1,6);
131  dswap(2,5);
132  dswap(3,4);
133  #undef dswap
134  #endif
135 }
